\chapter{Exercício 1}

Vamos discretizar uma das equações de Maxwell em 1 dimensão no vácuo, 
como por exemplo

\begin{equation}\label{eq:maxwell-1d-vacuo}
	\pd{E_y}{t} =  - \frac{1}{\epsilon} \pd{H_z}{z} 
\end{equation}

Usando diferenças centradas, cada uma das derivadas de (\ref{eq:maxwell-1d-vacuo}) 
pode ser aproximada por 

\[ \pd{E_y}{t} = \frac{E^{n+1} - E^n}{\Delta t} + O\left( \Delta t^2 \right) \quad , \quad 
	\pd{H_z}{z} = \frac{H_{i+1/2} - H_{i-1/2}}{\Delta x} + O\left( \Delta x^2 \right)
\]

\noindent onde as diferenças centradas resultam em erros de ordem 2. O erro $\tau_E$ é dado pela 
diferença entre os lados esquerdo e direito da Equação (\ref{eq:maxwell-1d-vacuo}) após substituir 
as aproximações, ou seja,

\begin{align*}
\tau_E
&=
\frac{E^{n+1} - E^n}{\Delta t} + O\left( \Delta t^2 \right) - \frac{1}{\epsilon} \left( \frac{H_{i+1/2} - H_{i-1/2}}{\Delta x} + O\left( \Delta x^2 \right) \right) 
\\[6pt]
&=
\left[ \frac{E^{n+1} - E^n}{\Delta t} -  \frac{1}{\epsilon} \frac{H_{i+1/2} - H_{i-1/2}}{\Delta x} \right] + O\left( \Delta t^2 \right) + \frac{1}{\epsilon} O\left( \Delta x^2 \right)
\\[6pt]
&=
\left[ \frac{E^{n+1} - E^n}{\Delta t} -  \frac{1}{\epsilon} \frac{H_{i+1/2} - H_{i-1/2}}{\Delta x} \right] + C \Delta t^2  + \frac{C}{\epsilon} \Delta x^2
\\[6pt]
&\text{Aplicando a Equação \ref{eq:maxwell-1d-vacuo} para substituir o termo entre colchetes, temos:}
\\[6pt]
\tau_E
&\leq
C \left( \Delta t^2 + \Delta x^2  \right)
\end{align*}

\begin{itemize}
	\item $N_x = N_y = 300$
	\item $\Delta x = \Delta y = 5 \un{mm}$
	\item $\Delta t = 10^{-12} \un{s}$
	\item $N_t$ = 2000
\end{itemize}

Note que esses parâmetros satisfazem a condição de estabilidade para o método de Yee em 2D,
uma vez que

\[  \Delta t \leq \frac{1}{c \sqrt{\frac{1}{(\Delta x)^2} + \frac{1}{(\Delta y)^2}}} \quad \Longrightarrow \quad 10^{-12} \leq 11.79 \cdot 10^{-12} \]
